    <script>
        // Academic typewriter effect
        const academicTerms = ['Visualization', 'Cartography', 'Analysis', 'Research'];
        let termIndex = 0;
        let charIndex = 0;
        let isDeleting = false;
        
        function academicTypewriter() {
            const current = academicTerms[termIndex];
            const typewriterElement = document.getElementById('typewriter');
            
            if (isDeleting) {
                typewriterElement.textContent = current.substring(0, charIndex - 1);
                charIndex--;
            } else {
                typewriterElement.textContent = current.substring(0, charIndex + 1);
                charIndex++;
            }
            
            if (!isDeleting && charIndex === current.length) {
                setTimeout(() => isDeleting = true, 3000);
            } else if (isDeleting && charIndex === 0) {
                isDeleting = false;
                termIndex = (termIndex + 1) % academicTerms.length;
            }
            
            setTimeout(academicTypewriter, isDeleting ? 80 : 120);
        }
        
        // Start typewriter after page load
        setTimeout(academicTypewriter, 1500);
        
        // Enhanced hover effects for chart panels
        document.querySelectorAll('.chart-panel').forEach(panel => {
            panel.addEventListener('mouseenter', () => {
                panel.style.transform = 'translateY(-3px) scale(1.01)';
            });
            
            panel.addEventListener('mouseleave', () => {
                panel.style.transform = 'translateY(0) scale(1)';
            });
        });
        
        // Subtle coordinate animation
        const coordElement = document.querySelector('.fixed.top-4.left-4');
        if (coordElement) {
            setInterval(() => {
                const coords = coordElement.children;
                const lat = coords[0];
                const lng = coords[1];
                
                // Subtle coordinate drift
                const latBase = 40 + (42 + Math.sin(Date.now() / 10000) * 0.1) / 60 + 46 / 3600;
                const lngBase = 74 + (0 + Math.cos(Date.now() / 8000) * 0.1) / 60 + 22 / 3600;
                
                const latDeg = Math.floor(latBase);
                const latMin = Math.floor((latBase - latDeg) * 60);
                const latSec = Math.floor(((latBase - latDeg) * 60 - latMin) * 60);
                
                const lngDeg = Math.floor(lngBase);
                const lngMin = Math.floor((lngBase - lngDeg) * 60);
                const lngSec = Math.floor(((lngBase - lngDeg) * 60 - lngMin) * 60);
                
                lat.textContent = `${latDeg}°${latMin}'${latSec}"N`;
                lng.textContent = `${lngDeg}°${lngMin}'${lngSec}"W`;
            }, 5000);
        }
